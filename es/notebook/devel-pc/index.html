syncthing

  curl -s https://syncthing.net/release-key.txt | sudo apt-key add -
  sudo sh -c 'echo deb http://apt.syncthing.net/ syncthing release > /etc/apt/sources.list.d/syncthing-release.list'
  sudo apt-get update
  sudo apt-get install syncthing
